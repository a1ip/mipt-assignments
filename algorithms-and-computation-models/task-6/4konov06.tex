$$
  \begin{pmatrix}
    1 & 0 & 0 & 5\\
    8 & 1 & 0 & 25\\
    4 & 8 & 1 & 125
  \end{pmatrix}
  \rightarrow
  \begin{pmatrix}
    1 & 0 & 0 & 5\\
    0 & 1 & 0 & -15\\
    0 & 8 & 1 & 5
  \end{pmatrix}
  \rightarrow
  \begin{pmatrix}
    1 & 0 & 0 & 5\\
    0 & 1 & 0 & -15\\
    0 & 0 & 1 & 125
  \end{pmatrix}
$$
